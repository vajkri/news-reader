Create `scripts/re-enrich.ts` as a standalone Node script. This is a one-time backfill script (per D-18) that re-enriches ALL remaining articles using the improved prompt and winning model.

**Key requirements:**
- Do NOT import from `enrich.ts` directly (it has `import 'server-only'` which fails in Node scripts). Instead, replicate the necessary logic.
- DO import from `@prisma/client` directly (create a local PrismaClient instance).
- DO import `generateText`, `Output` from `ai` and `z` from `zod`.
- Load `.env.local` for `DATABASE_URL` and `AI_GATEWAY_API_KEY`.

**Script structure:**

```typescript
import { config } from 'dotenv';
config({ path: '.env.local' });

import { generateText, Output } from 'ai';
import { z } from 'zod';
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();
const BATCH_SIZE = 15;

// Copy the AI_MODEL from src/lib/ai.ts (the winning model from Plan 03)
const AI_MODEL = 'WINNING_MODEL_STRING'; // <-- executor fills from ai.ts

// Copy the SEED_TOPICS from enrich.ts
const SEED_TOPICS = [
  'model releases', 'developer tools', 'industry moves',
  'research & breakthroughs', 'AI regulation & policy', 'open source', 'AI coding tools',
];

// Copy the improved SYSTEM_PROMPT from enrich.ts (post Plan 02)
const SYSTEM_PROMPT = `...`; // <-- executor copies from enrich.ts

// Copy ArticleEnrichmentSchema from enrich.ts
const ArticleEnrichmentSchema = z.object({
  articleId: z.number(),
  summary: z.string(),
  topics: z.array(z.string()),
  importanceScore: z.number().int().min(1).max(10),
  contentType: z.enum(['news', 'tutorial', 'opinion', 'announcement', 'roundup']),
  thinContent: z.boolean(),
});

type EnrichmentResult = z.infer<typeof ArticleEnrichmentSchema>;

function buildBatchPrompt(articles: { id: number; title: string; description: string | null; source: { name: string } }[]): string {
  return articles
    .map(a => `--- Article ID: ${a.id}\nSource: ${a.source.name}\nTitle: ${a.title}\nDescription: ${a.description ?? '(no description)'}`)
    .join('\n\n');
}

async function main(): Promise<void> {
  const totalCount = await prisma.article.count();
  console.log(`Total articles to re-enrich: ${totalCount}`);

  let offset = 0;
  let totalSaved = 0;
  let totalErrors = 0;
  let batchNum = 0;

  while (true) {
    const articles = await prisma.article.findMany({
      select: { id: true, title: true, description: true, source: { select: { name: true } } },
      orderBy: { publishedAt: 'desc' },
      skip: offset,
      take: BATCH_SIZE,
    });

    if (articles.length === 0) break;
    batchNum++;

    console.log(`\nBatch ${batchNum} (articles ${offset + 1}-${offset + articles.length} of ${totalCount}):`);

    try {
      const { output } = await generateText({
        model: AI_MODEL,
        output: Output.array({ element: ArticleEnrichmentSchema }),
        system: SYSTEM_PROMPT,
        prompt: buildBatchPrompt(articles),
        maxOutputTokens: 4096,
      });

      if (!output) {
        console.error(`  ERROR: AI returned no output for batch ${batchNum}`);
        totalErrors += articles.length;
        offset += BATCH_SIZE;
        continue;
      }

      const results: EnrichmentResult[] = output;
      const settled = await Promise.allSettled(
        results.map(result =>
          prisma.article.update({
            where: { id: result.articleId },
            data: {
              summary: result.summary,
              topics: result.topics,
              importanceScore: result.importanceScore,
              contentType: result.contentType,
              thinContent: result.thinContent,
              enrichedAt: new Date(),
            },
          })
        )
      );

      const saved = settled.filter(r => r.status === 'fulfilled').length;
      const errors = settled.filter(r => r.status === 'rejected').length;
      totalSaved += saved;
      totalErrors += errors;

      console.log(`  Saved: ${saved}, Errors: ${errors}`);
      if (errors > 0) {
        settled
          .filter((r): r is PromiseRejectedResult => r.status === 'rejected')
          .forEach(r => console.error(`    ${r.reason}`));
      }
    } catch (err) {
      console.error(`  BATCH ERROR: ${err instanceof Error ? err.message : String(err)}`);
      totalErrors += articles.length;
    }

    offset += BATCH_SIZE;
  }

  console.log(`\n--- Re-enrichment complete ---`);
  console.log(`Total saved: ${totalSaved}`);
  console.log(`Total errors: ${totalErrors}`);

  // Verify: count articles with null contentType (should be 0)
  const nullContentType = await prisma.article.count({ where: { contentType: null } });
  const nullThinContent = await prisma.article.count({ where: { thinContent: null } });
  console.log(`Articles with null contentType: ${nullContentType}`);
  console.log(`Articles with null thinContent: ${nullThinContent}`);

  await prisma.$disconnect();
}

main().catch(err => {
  console.error(err);
  process.exit(1);
});
```
