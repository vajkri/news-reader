    3. POST handler:
       ```typescript
       export async function POST(request: Request): Promise<Response> {
         const { messages } = await request.json();

         // Rate limiting (D-04, D-05)
         const ip = request.headers.get('x-forwarded-for') ?? 'anonymous';
         const retryAfter = await checkRateLimit(ip);
         if (retryAfter !== null) {
           return Response.json(
             { error: 'rate_limited', retryAfterMinutes: retryAfter },
             { status: 429 }
           );
         }
         await incrementRateLimit(ip);

         // Stream with tool-calling (D-01, D-02, D-03)
         const result = streamText({
           model: CHAT_MODEL,
           system: SYSTEM_PROMPT,
           messages: convertToModelMessages(messages.slice(-10)), // D-05: 10-turn history
           tools: {
             searchArticles: searchArticlesTool,
             articlesByTopic: articlesByTopicTool,
             recentArticles: recentArticlesTool,
           },
           stopWhen: stepCountIs(3), // tool-call -> tool-result -> final answer
         });

         return result.toUIMessageStreamResponse();
       }
       ```

    CRITICAL anti-patterns to avoid (from RESEARCH.md):
    - Do NOT use `generateText` (does not stream)
    - Do NOT use `toTextStreamResponse()` (loses tool call metadata)
    - Do NOT use `maxSteps` (removed in v6; use `stopWhen: stepCountIs(N)`)
    - Do NOT import `useChat` in this server file
    - Do NOT use `toDataStreamResponse()` (renamed in v6)
