1. Create `src/app/api/enrich/route.ts` with these exact contents:

```typescript
import { NextResponse } from 'next/server';
import {
  fetchUnenrichedArticles,
  enrichArticlesBatch,
  saveEnrichmentResults,
} from '@/lib/enrich';

export const maxDuration = 60;

export async function GET(request: Request): Promise<Response> {
  const authHeader = request.headers.get('authorization');
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return new Response('Unauthorized', { status: 401 });
  }

  const articles = await fetchUnenrichedArticles();

  if (articles.length === 0) {
    return NextResponse.json({ enriched: 0, skipped: 0, errors: [] });
  }

  try {
    const enrichmentResults = await enrichArticlesBatch(articles);
    const { saved, errors } = await saveEnrichmentResults(enrichmentResults);

    return NextResponse.json({
      enriched: saved,
      skipped: articles.length - saved,
      errors,
    });
  } catch (error) {
    return NextResponse.json(
      {
        enriched: 0,
        skipped: 0,
        errors: [error instanceof Error ? error.message : String(error)],
      },
      { status: 500 },
    );
  }
}
```

Key decisions:
- Uses GET (not POST) because Vercel cron jobs invoke via GET by default. The existing `/api/fetch` uses POST which may only work for manual triggers.
- `maxDuration = 60` gives enough time for AI API calls on Hobby plan (max 300s with Fluid Compute).
- Catches errors from the AI call itself and returns 500 with error message.
- Returns a structured JSON response with enriched count, skipped count, and errors array.
