1. Add `triggerEnrichment` server action to `src/lib/actions.ts`:

```typescript
export async function triggerEnrichment(): Promise<{ ok: boolean; enriched?: number; error?: string }> {
  const secret = process.env.CRON_SECRET;
  const appUrl = process.env.NEXT_PUBLIC_APP_URL ?? 'http://localhost:3000';

  if (!secret) {
    return { ok: false, error: 'CRON_SECRET not configured' };
  }

  try {
    const res = await fetch(`${appUrl}/api/enrich`, {
      headers: { Authorization: `Bearer ${secret}` },
    });
    if (!res.ok) {
      return { ok: false, error: `Enrichment failed: ${res.status}` };
    }
    const data = await res.json();
    return { ok: true, enriched: data.enriched };
  } catch (err) {
    return { ok: false, error: err instanceof Error ? err.message : 'Unknown error' };
  }
}
```

This runs server-side so CRON_SECRET is never exposed to the client (per Research Pitfall 7). Uses NEXT_PUBLIC_APP_URL for the base URL (already available), falls back to localhost for dev.

2. Create `src/components/features/briefing/EnrichNowButton.tsx`:
```tsx
'use client';

import { useState, useTransition } from 'react';
import { Button } from '@/components/ui/button';
import { useRouter } from 'next/navigation';
import { triggerEnrichment } from '@/lib/actions';

interface EnrichNowButtonProps {
  pendingCount: number;
}

export function EnrichNowButton({ pendingCount }: EnrichNowButtonProps): React.ReactElement {
  const [isPending, startTransition] = useTransition();
  const [error, setError] = useState(false);
  const router = useRouter();

  function handleEnrich() {
    setError(false);
    startTransition(async () => {
      const result = await triggerEnrichment();
      if (result.ok) {
        router.refresh();
      } else {
        setError(true);
        setTimeout(() => setError(false), 3000);
      }
    });
  }

  const label = error
    ? 'Enrichment failed. Try again.'
    : isPending
      ? 'Enriching...'
      : `Enrich now (${pendingCount})`;

  return (
    <Button
      variant="outline"
      size="sm"
      onClick={handleEnrich}
      disabled={isPending}
      className="whitespace-nowrap"
    >
      {label}
    </Button>
  );
}
```

Per D-08: User clicks to trigger, page reloads with results. No auto-enrichment.
Per UI-SPEC: Outline variant, loading state disabled, error resets after 3s.
Uses `useTransition` for non-blocking state update during server action.

interface PendingArticle {
